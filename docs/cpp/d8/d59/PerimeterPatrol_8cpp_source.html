161  // get waypoints
162  utility::SearchArea sa;
163  sa.from_container (*knowledge, region_id);
164  utility::Region reg = sa.get_convex_hull ();
165  vector<utility::GPSPosition> vertices = reg.vertices;
166 
167  // find closest waypoint as starting point
168  size_t closest = 0;
169  utility::GPSPosition current;
170  current.from_container (self_->device.location);
171  double min_distance = current.distance_to (vertices[0]);
172  for (size_t i = 1; i < vertices.size (); ++i)
173  {
174  double dist = current.distance_to (vertices[i]);
175  if (min_distance > dist)
176  {
177  dist = min_distance;
178  closest = i;
179  }
180  }
181 
182  // add some intermediate points
183  const size_t NUM_INTERMEDIATE_PTS = 5;
184  for (size_t i = 0; i < vertices.size(); ++i)
185  {
186  utility::GPSPosition start = vertices[(i + closest) % vertices.size ()];
187  utility::GPSPosition end = vertices[(i + closest + 1) % vertices.size ()];
188  double lat_dif = start.latitude () - end.latitude ();
189  double lon_dif = start.longitude () - end.longitude ();
190 
191  for (size_t j = 0; j < NUM_INTERMEDIATE_PTS; ++j)
192  {
193  utility::GPSPosition temp (
194  start.latitude () - lat_dif * j / NUM_INTERMEDIATE_PTS,
195  start.longitude () - lon_dif * j / NUM_INTERMEDIATE_PTS,
196  self_->device.desired_altitude.to_double ());
197  waypoints_.push_back (temp);
198  }
199  }
200 
201  // set next_position_
