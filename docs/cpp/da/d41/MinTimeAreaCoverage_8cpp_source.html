297 double
298 gams::algorithms::area_coverage::MinTimeAreaCoverage::get_utility (
299  const utility::Position& start, const utility::Position& end,
300  std::set<utility::Position>& online)
301 {
306  double util = 0.0;
307  const double radius =
308  min_time_.get_range () / min_time_.get_discretization ();
309  for (std::set<utility::Position>::const_iterator it = valid_positions_.begin ();
310  it != valid_positions_.end (); ++it)
311  {
312  if (start.distance_to_2d (end, *it) < radius)
313  {
314  double time = min_time_.get_value (*it);
315  double delta_util = pow (time, 3.0);
316  util += delta_util;
317  online.insert (*it);
318  }
319  }
320 
321  // modify the utility based on the distance that will be travelled
322  return util / sqrt(start.distance_to_2d (end) + 1);
323 }
