292 void
293 gams::platforms::VREP_Base::gps_to_vrep (const utility::GPS_Position & position,
294  utility::Position & converted) const
295 {
296  // assume the Earth is a perfect sphere
297  const double EARTH_RADIUS = 6371000.0;
298  const double EARTH_CIRCUMFERENCE = 2 * EARTH_RADIUS * M_PI;
299 
300  // convert the latitude/y coordinates
301  converted.y = (position.latitude () - sw_position_.latitude ()) / 360.0 * EARTH_CIRCUMFERENCE;
302 
303  // assume the meters/degree longitude is constant throughout environment
304  // convert the longitude/x coordinates
305  double r_prime = EARTH_RADIUS * cos (DEG_TO_RAD (sw_position_.latitude ()));
306  double circumference = 2 * r_prime * M_PI;
307  converted.x = (position.longitude () - sw_position_.longitude ()) / 360.0 * circumference;
308 
309  // do nothing to altitude
310  converted.z = position.altitude ();
311 }

323 gams::platforms::VREP_Base::vrep_to_gps (const utility::Position & position,
324  utility::GPS_Position & converted) const
325 {
326  // assume the Earth is a perfect sphere
327  const double EARTH_RADIUS = 6371000.0;
328  const double EARTH_CIRCUMFERENCE = 2 * EARTH_RADIUS * M_PI;
329 
330  // convert the latitude/y coordinates
331  // VREP uses y for latitude
332  converted.latitude ((360.0 * position.y / EARTH_CIRCUMFERENCE) + sw_position_.latitude ());
333 
334  // assume the meters/degree longitude is constant throughout environment
335  // convert the longitude/x coordinates
336  // VREP uses x for longitude
337  double r_prime = EARTH_RADIUS * cos (DEG_TO_RAD (sw_position_.latitude ()));
338  double circumference = 2 * r_prime * M_PI;
339  converted.longitude ((360.0 * position.x / circumference) + sw_position_.longitude ());
340 
341  // do nothing to altitude
342  converted.altitude (position.z);
343 }
