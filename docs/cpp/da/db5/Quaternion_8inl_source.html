111  inline void Quaternion::from_orientation_vector(
112  double rx, double ry, double rz)
113  {
114  double magnitude = sqrt(rx * rx + ry * ry + rz * rz);
115  if(magnitude == 0)
116  {
117  w_ = 1;
118  x_ = y_ = z_ = 0;
119  }
120  else
121  {
122  double half_mag = magnitude / 2;
123  double cos_half_mag = cos(half_mag);
124  double sin_half_mag = sin(half_mag);
125  w_ = cos_half_mag;
126  x_ = (rx / magnitude) * sin_half_mag;
127  y_ = (ry / magnitude) * sin_half_mag;
128  z_ = (rz / magnitude) * sin_half_mag;
129  }
130  }

137  inline void Quaternion::to_orientation_vector(
138  double &rx, double &ry, double &rz) const
139  {
140  double norm = sqrt(x_ * x_ + y_ * y_ + z_ * z_);
141  double angle = 2 * atan2(norm, w_);
142  double sin_half_angle = sin(angle / 2);
143  if(sin_half_angle < 1e-10)
144  {
145  rx = ry = rz = 0;

147  else
148  {
149  rx = (x_ / sin_half_angle) * angle;
150  ry = (y_ / sin_half_angle) * angle;
151  rz = (z_ / sin_half_angle) * angle;
152  }

160  inline void Quaternion::hamilton_product(
161  Quaternion &into, const Quaternion &lhs, const Quaternion &rhs)
162  {
163  double a = (lhs.w_ + lhs.x_) * (rhs.w_ + rhs.x_),
164  b = (lhs.z_ - lhs.y_) * (rhs.y_ - rhs.z_),
165  c = (lhs.w_ - lhs.x_) * (rhs.y_ + rhs.z_),
166  d = (lhs.y_ + lhs.z_) * (rhs.w_ - rhs.x_),
167  e = (lhs.x_ + lhs.z_) * (rhs.x_ + rhs.y_),
168  f = (lhs.x_ - lhs.z_) * (rhs.x_ - rhs.y_),
169  g = (lhs.w_ + lhs.y_) * (rhs.w_ - rhs.z_),
170  h = (lhs.w_ - lhs.y_) * (rhs.w_ + rhs.z_);
171 
172  into.w_ = b + (-e - f + g + h) / 2;
173  into.x_ = a - ( e + f + g + h) / 2;
174  into.y_ = c + ( e - f + g - h) / 2;
175  into.z_ = d + ( e - f - g + h) / 2;
176  }

286  {
