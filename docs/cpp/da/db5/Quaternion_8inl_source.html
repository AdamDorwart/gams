109  inline void Quaternion::from_rotation_vector(
110  double rx, double ry, double rz)
111  {
112  double magnitude = sqrt(rx * rx + ry * ry + rz * rz);
113  if(magnitude == 0)
114  {
115  w_ = 1;
116  x_ = y_ = z_ = 0;
117  }
118  else
119  {
120  double half_mag = magnitude / 2;
121  double cos_half_mag = cos(half_mag);
122  double sin_half_mag = sin(half_mag);
123  w_ = cos_half_mag;
124  x_ = (rx / magnitude) * sin_half_mag;
125  y_ = (ry / magnitude) * sin_half_mag;
126  z_ = (rz / magnitude) * sin_half_mag;
127  }
128  }

135  inline void Quaternion::to_rotation_vector(
136  double &rx, double &ry, double &rz) const
137  {
138  double norm = sqrt(x_ * x_ + y_ * y_ + z_ * z_);
139  double angle = 2 * atan2(norm, w_);
140  double sin_half_angle = sin(angle / 2);
141  if(sin_half_angle < 1e-10)
142  {
143  rx = ry = rz = 0;
144  }
145  else
146  {
147  rx = (x_ / sin_half_angle) * angle;
148  ry = (y_ / sin_half_angle) * angle;
149  rz = (z_ / sin_half_angle) * angle;
150  }

158  inline void Quaternion::hamilton_product(
159  Quaternion &into, const Quaternion &lhs, const Quaternion &rhs)
160  {
161  double a = (lhs.w_ + lhs.x_) * (rhs.w_ + rhs.x_),
162  b = (lhs.z_ - lhs.y_) * (rhs.y_ - rhs.z_),
163  c = (lhs.w_ - lhs.x_) * (rhs.y_ + rhs.z_),
164  d = (lhs.y_ + lhs.z_) * (rhs.w_ - rhs.x_),
165  e = (lhs.x_ + lhs.z_) * (rhs.x_ + rhs.y_),
166  f = (lhs.x_ - lhs.z_) * (rhs.x_ - rhs.y_),
167  g = (lhs.w_ + lhs.y_) * (rhs.w_ - rhs.z_),
168  h = (lhs.w_ - lhs.y_) * (rhs.w_ + rhs.z_);
169 
170  into.w_ = b + (-e - f + g + h) / 2;
171  into.x_ = a - ( e + f + g + h) / 2;
172  into.y_ = c + ( e - f + g - h) / 2;
173  into.z_ = d + ( e - f - g + h) / 2;
174  }
