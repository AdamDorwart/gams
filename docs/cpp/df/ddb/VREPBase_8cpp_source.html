367  gams::loggers::LOG_TRACE,
368  "gams::platforms::VREPBase::do_move:" \
369  " moving from (%f, %f, %f) to (%f, %f, %f, distance %f m).",
370  current.x (), current.y (), current.z (),
371  target.x (), target.y (), target.z (), distance);
372 
373  if (max_delta > 0)
374  {
375  // move quadrotor target closer to the desired position
376  if (distance < max_delta) // we can get to target in one step
377  {
378  madara_logger_ptr_log (gams::loggers::global_logger.get (),
379  gams::loggers::LOG_TRACE,
380  "gams::platforms::VREPBase::do_move:" \

382 
383  curr_arr[0] = dest_arr[0];
384  curr_arr[1] = dest_arr[1];
385  curr_arr[2] = dest_arr[2];
386  }
387  else // we cannot reach target in this step
388  {
389  madara_logger_ptr_log (gams::loggers::global_logger.get (),
390  gams::loggers::LOG_TRACE,

393 
394  // how far do we have to go in each dimension
395  double dist[3];

397  dist[i] = curr_arr[i] - dest_arr[i];
398 
399  // update target position
400  for (int i = 0; i < 3; ++i)
401  {
402  curr_arr[i] -= dist[i] * max_delta / distance;
403  }
404  }
405
