304 algorithms::AlgorithmFactoryRepository::create (
305  const std::string & type,
306  const madara::knowledge::KnowledgeMap & args)
307 {
308  algorithms::BaseAlgorithm * result = 0;
309 
310  // the user is going to expect this kind of error to be printed immediately
311  madara_logger_ptr_log (gams::loggers::global_logger.get (),
312  gams::loggers::LOG_MAJOR,
313  "gams::algorithms::AlgorithmFactoryRepository::create:" \
314  " creating \"%s\" algorithm.\n", type.c_str ());
315 
316 
317  if (type != "")
318  {
319  std::string lowercased_type (type);
320  madara::utility::lower (lowercased_type);
321 
322  AlgorithmFactoryMap::iterator it = factory_map_.find (lowercased_type);
323  if (it != factory_map_.end ())
324  {
325  result = it->second->create (args, knowledge_, platform_,
326  sensors_, self_, agents_);
327  }
328  else
329  {
330  // the user is going to expect this kind of error to be printed immediately
331  madara_logger_ptr_log (gams::loggers::global_logger.get (),
332  gams::loggers::LOG_ALWAYS,
333  "gams::algorithms::AlgorithmFactoryRepository::create:" \
334  " could not find \"%s\" algorithm.\n", lowercased_type.c_str ());
335  }
336  }
337 
338  return result;
339 }
