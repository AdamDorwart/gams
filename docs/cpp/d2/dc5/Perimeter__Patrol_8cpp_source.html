167  utility::GPS_Position current;
168  current.from_container (self_->device.location);
169  double min_distance = current.distance_to (vertices[0]);
170  for (size_t i = 1; i < vertices.size (); ++i)
171  {
172  double dist = current.distance_to (vertices[i]);
173  if (min_distance > dist)
174  {
175  dist = min_distance;
176  closest = i;
177  }
178  }
179 
180  // add some intermediate points

182  for (size_t i = 0; i < vertices.size(); ++i)
183  {
184  utility::GPS_Position start = vertices[(i + closest) % vertices.size ()];
185  utility::GPS_Position end = vertices[(i + closest + 1) % vertices.size ()];
186  double lat_dif = start.latitude () - end.latitude ();
187  double lon_dif = start.longitude () - end.longitude ();
188 
189  for (size_t j = 0; j < NUM_INTERMEDIATE_PTS; ++j)
190  {
191  utility::GPS_Position temp (
192  start.latitude () - lat_dif * j / NUM_INTERMEDIATE_PTS,
193  start.longitude () - lon_dif * j / NUM_INTERMEDIATE_PTS,
194  self_->device.desired_altitude.to_double ());
195  waypoints_.push_back (temp);
196  }
197  }
