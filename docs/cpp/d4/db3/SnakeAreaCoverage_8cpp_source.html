235 gams::algorithms::area_coverage::SnakeAreaCoverage::compute_waypoints (
236  const string& region_id)
237 {
238  if (platform_ && *platform_->get_platform_status ()->movement_available)
239  {
240  // get region information
241  utility::Region region;
242  region.from_container (*knowledge_, region_id);
243 
244  madara_logger_ptr_log (gams::loggers::global_logger.get (),
245  gams::loggers::LOG_DETAILED,
246  "gams::algorithms::SnakeAreaCoverage::compute_waypoints:" \
247  " using region \"%s\"\n", region.to_string ().c_str ());
248 
249  // convert to equirectangular projection coordinates
250  const size_t num_edges = region.vertices.size ();
251  vector<utility::Position> positions;
252  const utility::GPSPosition reference = region.vertices[0];
253  for (size_t i = 0; i < region.vertices.size (); ++i)
254  positions.push_back (region.vertices[i].to_position (reference));
255 
256  // find longest edge
257  size_t longest_edge = 0;
258  double max_dist = positions[0].distance_to_2d (positions[1]);
259  for (size_t i = 1; i < num_edges; ++i)
260  {
261  double dist = positions[i].distance_to_2d (
262  positions[(i + 1) % num_edges]);
263  if (dist > max_dist)
264  {
265  max_dist = dist;
266  longest_edge = i;
267  }
268  }
269 
270  // starting points are vertices of longest edge
271  utility::GPSPosition temp = utility::GPSPosition::to_gps_position (
272  positions[longest_edge], reference);
273  temp.altitude (self_->agent.desired_altitude.to_double ());
274  waypoints_.push_back (temp);
275  temp = utility::GPSPosition::to_gps_position (
276  positions[(longest_edge + 1) % num_edges], reference);
277  temp.altitude (self_->agent.desired_altitude.to_double ());
278  waypoints_.push_back (temp);
279 
280  // determine shift direction
281  const double shift = 2.5; // distance in meters between parallel lines
282  const utility::Position p_0 = positions[longest_edge];
283  const utility::Position p_1 = positions[(longest_edge + 1) % num_edges];
284 
292  for (int dir = 0; dir < 2; ++dir)
293  {
294  // loop until no more intercepts are found
295  int intercept_idx = -1; // arbitrary non-zero value so we enter loop
296  unsigned int loop = 0; // loop counter
297  while (intercept_idx != 0)
298  {
299  intercept_idx = 0;
300  ++loop;
301 
307  utility::Position intercepts[2];
308  for (size_t i = 1; i < num_edges && intercept_idx < 2; ++i)
309  {
310  // check current vertex for intercept
311  int cur_vertex = (longest_edge + i) % (int)num_edges;
312 
313  // beginning and end vertex of intersecing line segment
314  const utility::Position p_n_0 = positions[cur_vertex];
315  const utility::Position p_n_1 =
316  positions[(cur_vertex + 1) % num_edges];
317 
318  double m_0, m_n;
319  utility::Position check;
320  bool check_intercept = true;
321  // if longest_edge is not vertical
322  if (p_0.slope_2d (p_1, m_0))
323  {
324  // we have a non-vertical line, so find new intercept
325  const double delta_b = pow (-1.0, dir) * shift *
326  pow (pow (m_0, 2) + 1, 0.5);
327 
328  // calculate potential waypoint
329  if (p_n_0.slope_2d (p_n_1, m_n)) // potential edge not a vertical line
330  {
331  if (m_n != m_0) // parallel lines can't intersect
332  {
333  // find intercept of a line parallel to longest_edge and edge
334  // we are checking
335  check.y = (m_n * p_0.y - m_n * m_0 * p_0.x +
336  m_n * loop * delta_b - m_0 * p_n_0.y + m_0 * m_n * p_n_0.x) /
337  (m_n - m_0);
338  check.x = (check.y - p_n_0.y + m_n * p_n_0.x) / m_n;
339  }
340  else // edges are parallel
341  check_intercept = false;
342  }
343  else // edge to check is vertical, perform slopeless waypoint calculation
344  {
345  check.x = p_n_0.x; // vertical line has same x coord throughout
346  check.y = m_0 * check.x + p_0.y - m_0 * p_0.x + loop * delta_b;
347  }
348  } // end if edge to check is not vertical
349  else // longest_edge is vertical line, so just shift the x coord
350  {
355  if (p_n_0.slope_2d (p_n_1, m_n))
356  {
357  // longest edge is vertical, so just shift the x coord
358  check.x = p_0.x + pow (-1.0, (double)dir) * loop * shift;
359  check.y = m_n * (check.x - p_n_0.x) + p_n_0.y;
360  }
361  else
362  check_intercept = false;
363  } // end else longest_edge is vertical line
364 
365  // check if it's actually an intercept
366  check.z = self_->agent.desired_altitude.to_double (); // TODO: actual altitude control
367  if (check_intercept && p_n_0.is_between_2d (p_n_1, check))
368  intercepts[intercept_idx++] = check;
369  } // end foreach edge
370 
371  // found intercepts => go to closest one first
372  if (intercept_idx > 1)
373  {
374  const utility::Position prev =
375  waypoints_[waypoints_.size () - 1].to_position (reference);
376  if (prev.distance_to_2d (intercepts[0]) >
377  prev.distance_to_2d (intercepts[1]))
378  {
379  waypoints_.push_back (
380  utility::GPSPosition::to_gps_position (intercepts[1], reference));
381  waypoints_.push_back (
382  utility::GPSPosition::to_gps_position (intercepts[0], reference));
383  }
384  else
385  {
386  waypoints_.push_back (
387  utility::GPSPosition::to_gps_position (intercepts[0], reference));
388  waypoints_.push_back (
389  utility::GPSPosition::to_gps_position (intercepts[1], reference));
390  }
391  }
392  else if (intercept_idx > 0)
393  {
394  waypoints_.push_back (
395  utility::GPSPosition::to_gps_position (intercepts[0], reference));
396  }
397  } // end while still finding intercepts
398  } // end for +/- delta_b
399 
400  size_t idx = 0;
401  for (const utility::GPSPosition& p : waypoints_)
402  {
403  madara_logger_ptr_log (gams::loggers::global_logger.get (),
404  gams::loggers::LOG_DETAILED,
