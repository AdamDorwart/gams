380 int
381 gams::algorithms::Formation_Flying::plan (void)
382 {
383  // increment executions, only used by rotation formation for now
384  ++executions_;
385 
386  need_to_move_ = false;
387  if (head_)
388  {
389  // head only has to wait for everybody, and then move to destination
390  if (formation_ready_ == 1)
391  {
392  madara_logger_ptr_log (gams::loggers::global_logger.get (),
393  gams::loggers::LOG_DETAILED,
394  "gams::algorithms::Formation_Flying:" \
395  " head is getting destination\n");
396  next_position_ = get_destination ();
397  need_to_move_ = true;
398  }
399  else
400  {
401  madara_logger_ptr_log (gams::loggers::global_logger.get (),
402  gams::loggers::LOG_DETAILED,
403  "gams::algorithms::Formation_Flying:" \
404  " head formation not ready\n");
405  }
406  }
407  else // !head_
408  {
409  switch (modifier_)
410  {
414  case ROTATE:
415  {
416  const double OMEGA = M_PI / 30;
417  double angle = -phi_ + phi_dir_ + executions_ * OMEGA;
418  utility::Position offset (rho_ * cos (angle), rho_ * sin (angle), z_);
419 
420  utility::GPS_Position reference;
421  reference.from_container (head_location_);
422  next_position_ = utility::GPS_Position::to_gps_position (
423  offset, reference);
424 
425  need_to_move_ = true;
426 
427  break;
428  }
429 
433  default: // case NONE
434  {
435  // calculate formation location
436  double angle = phi_ + phi_dir_;
437  utility::GPS_Position ref_location;
438  ref_location.from_container (head_location_);
439  utility::Position offset (rho_ * cos (angle), rho_ * sin (angle), z_);
440 
441  // hold position until everybody is ready
442  if (formation_ready_ == 0)
443  {
444  madara_logger_ptr_log (gams::loggers::global_logger.get (),
445  gams::loggers::LOG_DETAILED,
446  "gams::algorithms::Formation_Flying:" \
447  " follower moving or holding position in formation\n");
448  next_position_ = utility::GPS_Position::to_gps_position (
449  offset, ref_location);
450  }
451  else // we are moving or already at destination
452  {
453  madara_logger_ptr_log (gams::loggers::global_logger.get (),
454  gams::loggers::LOG_DETAILED,
455  "gams::algorithms::Formation_Flying:" \
456  " follower moving to destination\n");
457  double dist = ref_location.distance_to (get_destination ());
458  // TODO: tune the movement parameter
459  if (dist > platform_->get_move_speed () * 1.5)
460  {
461  // predict where the reference device will be
462  dist = 1;
463  utility::Position direction (
464  dist * cos (phi_dir_), dist * sin (phi_dir_));
465  utility::GPS_Position predicted =
466  utility::GPS_Position::to_gps_position (
467  direction, ref_location);
468  next_position_ = utility::GPS_Position::to_gps_position (
469  offset, predicted);
470  }
471  else // close enough, just go to final location
472  {
473  next_position_ = utility::GPS_Position::to_gps_position (
474  offset, get_destination ());
475  }
476  }
477  need_to_move_ = true;
478  }
479  }
480  }
481  return 0;
482 }
