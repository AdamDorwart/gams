459 int
460 gams::algorithms::Formation_Flying::plan (void)
461 {
462  // increment executions, only used by rotation formation for now
463  ++executions_;
464 
465  need_to_move_ = false;
466  if (head_)
467  {
468  // head only has to wait for everybody, and then move to destination
469  if (formation_ready_ == 1)
470  {
471  madara_logger_ptr_log (gams::loggers::global_logger.get (),
472  gams::loggers::LOG_DETAILED,
473  "gams::algorithms::Formation_Flying:" \
474  " head is getting destination\n");
475  next_position_ = get_destination ();
476  need_to_move_ = true;
477  }
478  else
479  {
480  madara_logger_ptr_log (gams::loggers::global_logger.get (),
481  gams::loggers::LOG_DETAILED,
482  "gams::algorithms::Formation_Flying:" \
483  " head formation not ready\n");
484  }
485  }
486  else // !head_
487  {
488  switch (modifier_)
489  {
493  case ROTATE:
494  {
495  const double OMEGA = M_PI / 30;
496  double angle = -phi_ + phi_dir_ + executions_ * OMEGA;
497  utility::Position offset (rho_ * cos (angle), rho_ * sin (angle), z_);
498 
499  utility::GPS_Position reference;
500  reference.from_container (head_location_);
501  next_position_ = utility::GPS_Position::to_gps_position (
502  offset, reference);
503 
504  need_to_move_ = true;
505 
506  break;
507  }
508 
512  default: // case NONE
513  {
514  // calculate formation location
515  double angle = phi_ + phi_dir_;
516  utility::GPS_Position ref_location;
517  ref_location.from_container (head_location_);
518  utility::Position offset (rho_ * cos (angle), rho_ * sin (angle), z_);
519 
520  // hold position until everybody is ready
521  if (formation_ready_ == 0)
522  {
523  madara_logger_ptr_log (gams::loggers::global_logger.get (),
524  gams::loggers::LOG_DETAILED,
525  "gams::algorithms::Formation_Flying:" \
526  " follower moving or holding position in formation\n");
527  next_position_ = utility::GPS_Position::to_gps_position (
528  offset, ref_location);
529  }
530  else // we are moving or already at destination
531  {
532  madara_logger_ptr_log (gams::loggers::global_logger.get (),
533  gams::loggers::LOG_DETAILED,
534  "gams::algorithms::Formation_Flying:" \
535  " follower moving to destination\n");
536  double dist = ref_location.distance_to (get_destination ());
537  // TODO: tune the movement parameter
538  if (dist > platform_->get_move_speed () * 1.5)
539  {
540  // predict where the reference device will be
541  dist = 1;
542  utility::Position direction (
543  dist * cos (phi_dir_), dist * sin (phi_dir_));
544  utility::GPS_Position predicted =
545  utility::GPS_Position::to_gps_position (
546  direction, ref_location);
547  next_position_ = utility::GPS_Position::to_gps_position (
548  offset, predicted);
549  }
550  else // close enough, just go to final location
551  {
552  next_position_ = utility::GPS_Position::to_gps_position (
553  offset, get_destination ());
554  }
555  }
556  need_to_move_ = true;
557  }
558  }
559  }
560  return 0;
561 }
