516 gams::controllers::BaseController::run (double loop_period,
517  double max_runtime, double send_period)
518 {
519  // return value
520  int return_value (0);
521  bool first_execute (true);
522 
523  // for checking for potential user commands
524  double loop_hz = 1.0 / loop_period;
525  double send_hz = 1.0 / send_period;
526 
527  self_.agent.loop_hz = loop_hz;
528  self_.agent.send_hz = send_hz;
529 
530  // if user specified non-positive, then we are to use loop_period
531  if (send_period <= 0)
532  {
533  send_period = loop_period;
534  }
535 
536  // loop every period until a max run time has been reached
537  ACE_Time_Value current = ACE_OS::gettimeofday ();
538  ACE_Time_Value max_wait, sleep_time, next_epoch;
539  ACE_Time_Value send_sleep_time, send_next_epoch;

542 
543  madara_logger_ptr_log (gams::loggers::global_logger.get (),
544  gams::loggers::LOG_MAJOR,
545  "gams::controllers::BaseController::run:" \
546  " loop_period: %fs, max_runtime: %fs, send_period: %fs\n",

548 
549  madara_logger_ptr_log (gams::loggers::global_logger.get (),
550  gams::loggers::LOG_MAJOR,

552  " calling system_analyze ()\n");
553  return_value |= system_analyze ();
554 
555  if (loop_period >= 0.0)
556  {
557  max_wait.set (max_runtime);
558  max_wait = current + max_wait;
559 
560  poll_frequency.set (loop_period);
561  send_poll_frequency.set (send_period);
562  next_epoch = current + poll_frequency;
563  send_next_epoch = current;
564 
565  //unsigned int iterations = 0;
566  while (first_execute || max_runtime < 0 || current < max_wait)

575  return_value |= system_analyze ();
576 
577  // grab current time
578  current = ACE_OS::gettimeofday ();
579 
580  // run will always try to send at least once
581  if (first_execute || current > send_next_epoch)
582  {
583  madara_logger_ptr_log (gams::loggers::global_logger.get (),
584  gams::loggers::LOG_MAJOR,

589  knowledge_.send_modifieds ();
590 
591  // setup the next send epoch
592  if (send_period > 0)

595  send_next_epoch += send_poll_frequency;
596  }
597  }
598
