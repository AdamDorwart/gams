135 struct sort_by_angle
136 {
137  const gams::utility::GPS_Position anchor;
138 
139  sort_by_angle (const gams::utility::GPS_Position& p) : anchor (p) {}
140 
141  bool operator() (const gams::utility::GPS_Position& gp1,
142  const gams::utility::GPS_Position& gp2)
143  {
144  gams::utility::Position p1 = gp1.to_position (anchor);
145  gams::utility::Position p2 = gp2.to_position (anchor);
146 
147  double angle1 = atan2 (p1.x, p1.y) + 2 * M_PI;
148  double angle2 = atan2 (p2.x, p2.y) + 2 * M_PI;
149 
150  if (angle1 == angle2)
151  return (anchor.distance_to(gp1) < anchor.distance_to(gp2));
152  else
153  return (angle1 < angle2);
154  }

158 gams::utility::Search_Area::get_convex_hull () const
159 {
165  // get all points, filter out duplicates
166  set<GPS_Position> s_points;
167  for (size_t i = 0; i < regions_.size (); ++i)
168  for (size_t j = 0; j < regions_[i].vertices.size (); ++j)
169  s_points.insert (regions_[i].vertices[j]);
170  const size_t N = s_points.size ();
171 
172  // create array of points
173  vector <GPS_Position> points (N + 1);
174  vector <GPS_Position>::iterator start = points.begin ();
175  ++start;
176  copy (s_points.begin (), s_points.end (), start);
177 // for (int i = 0; i < N+1; ++i)
178 // cerr << std::setprecision(10) << points[i].latitude() << " " << points[i].longitude() << endl;
179 
180  // find point with lowest y/lat coord...
181  size_t lowest = 1;
182  double min_lat = points[1].latitude ();
183  for (size_t i = 2; i <= N; ++i)
184  {
185  if (points[i].latitude () < min_lat)
186  {
187  lowest = i;
188  min_lat = points[i].latitude ();
189  }
190  else if (points[i].latitude () == min_lat)
191  {
192  // select western-most point
193  if (points[i].longitude () < points[lowest].longitude ())
194  lowest = i;
195  }
196  }
197 
198  // ...and swap with points[1]
199  swap (points[lowest], points[1]);
200  //cerr << "selected " << points[1].latitude() << " " << points[1].longitude() << " as lowest" << endl;
201 
202  // sort positions
203  sort (&points[2], &points[N + 1], sort_by_angle (points[1]));
204 // cerr << "sorting points" << endl;
205 // for (int i = 0; i < N+1; ++i)
206 // cerr << points[i].latitude() << " " << points[i].longitude() << endl;
207 
208  // copy sentinel point
209  points[0] = points[N];
210 
211  // find convex hull
212  unsigned int M = 1;
213  for (unsigned int i = 2; i <= N; ++i)
214  {
215  // find next valid point on convex hull
216  while (cross (points[M - 1], points[M], points[i]) <= 0)
217  {
218  if (M > 1)
219  M -= 1;
220  else if (i == N)
221  break;
222  else
223  i += 1;
224  }
225 
226  // update M and swap points to the correct place
227  M += 1;
228  swap (points[M], points[i]);
229  }
230 
231  // fill vector of points
232  vector<GPS_Position> temp (M);
233  copy (&points[1], &points[M + 1], temp.begin ());
234 // for (vector<GPS_Position>::iterator it = temp.begin(); it != temp.end(); ++it)
235 // cerr << "\t" << it->latitude () << " " << it->longitude () << endl;
236  return Region (temp);
237 }
