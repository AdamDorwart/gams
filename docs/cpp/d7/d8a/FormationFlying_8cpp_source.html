501  // increment executions, only used by rotation formation for now
502  ++executions_;
503 
504  need_to_move_ = false;
505  if (head_)
506  {
507  // head only has to wait for everybody, and then move to destination
508  if (formation_ready_ == 1)

511  gams::loggers::LOG_DETAILED,

520  gams::loggers::LOG_DETAILED,
521  "gams::algorithms::FormationFlying:" \
522  " head formation not ready\n");
523  }
524  }
525  else // !head_
526  {
527  switch (modifier_)
528  {
532  case ROTATE:
533  {
534  const double OMEGA = M_PI / 30;
535  double angle = -phi_ + phi_dir_ + executions_ * OMEGA;
536  utility::Position offset (rho_ * cos (angle), rho_ * sin (angle), z_);
537 
538  utility::GPSPosition reference;

540  next_position_ = utility::GPSPosition::to_gps_position (
541  offset, reference);
542 
543  need_to_move_ = true;
544 
545  break;
546  }
547 
551  default: // case NONE
552  {
553  // calculate formation location
554  double angle = phi_ + phi_dir_;
555  utility::GPSPosition ref_location;
556  ref_location.from_container (head_location_);
557  utility::Position offset (rho_ * cos (angle), rho_ * sin (angle), z_);
558 
559  // hold position until everybody is ready

563  gams::loggers::LOG_DETAILED,

568  }

572  gams::loggers::LOG_DETAILED,
