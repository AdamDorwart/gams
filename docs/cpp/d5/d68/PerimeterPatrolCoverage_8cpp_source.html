198 gams::algorithms::area_coverage::PerimeterPatrolCoverage::generate_positions (void)
199 {
200  if (platform_ && *platform_->get_platform_status ()->movement_available)
201  {
202  // get waypoints
203  utility::SearchArea sa;
204  sa.from_container (*knowledge_, region_id_);
205  utility::Region reg = sa.get_convex_hull ();
206  vector<utility::GPSPosition> vertices = reg.vertices;
207 
208  // find closest waypoint as starting point
209  size_t closest = 0;
210  utility::GPSPosition current;
211  current.from_container (self_->agent.location);
212  double min_distance = current.distance_to (vertices[0]);
213  for (size_t i = 1; i < vertices.size (); ++i)
214  {
215  double dist = current.distance_to (vertices[i]);
216  if (min_distance > dist)
217  {
218  dist = min_distance;
219  closest = i;
220  }
221  }
222 
223  // add some intermediate points
224  const size_t NUM_INTERMEDIATE_PTS = 1;
225  for (size_t i = 0; i < vertices.size (); ++i)
226  {
227  utility::GPSPosition start = vertices[(i + closest) % vertices.size ()];
228  utility::GPSPosition end = vertices[(i + closest + 1) % vertices.size ()];
229  double lat_dif = start.latitude () - end.latitude ();
230  double lon_dif = start.longitude () - end.longitude ();
231 
232  for (size_t j = 0; j < NUM_INTERMEDIATE_PTS; ++j)
233  {
234  utility::GPSPosition temp (
235  start.latitude () - lat_dif * j / NUM_INTERMEDIATE_PTS,
236  start.longitude () - lon_dif * j / NUM_INTERMEDIATE_PTS,
237  self_->agent.desired_altitude.to_double ());
238  waypoints_.push_back (temp);
239  }
240  }
241 
242  // set next_position_
243  cur_waypoint_ = 0;
244  next_position_ = waypoints_[cur_waypoint_];
245 
246  initialized_ = true;
247  }
