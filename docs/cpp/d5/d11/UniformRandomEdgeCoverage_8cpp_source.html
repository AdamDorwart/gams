219 gams::algorithms::area_coverage::UniformRandomEdgeCoverage::generate_new_position ()
220 {
221  // select new edge
222  int num_edges = int (region_.vertices.size());
223  int target_edge = madara::utility::rand_int (0, num_edges-1);
224 
225  // get endpoints
226  const utility::GPSPosition & pos_1 = region_.vertices[target_edge];
227  const utility::GPSPosition & pos_2 =
228  region_.vertices[(target_edge + 1) % num_edges];
229 
230  // get random point on line
231  double delta_lat = pos_2.latitude () - pos_1.latitude ();
232  double delta_lon = pos_2.longitude () - pos_1.longitude ();
233  if (delta_lon == 0) // north/south line
234  {
235  const double & min = pos_1.latitude () < pos_2.latitude () ? pos_1.latitude () : pos_2.latitude ();
236  const double & max = pos_1.latitude () > pos_2.latitude () ? pos_1.latitude () : pos_2.latitude ();
237  next_position_.latitude (madara::utility::rand_double(min, max));
238  next_position_.longitude (pos_1.longitude ());
239  }
240  else if (delta_lat == 0) // east/west line
241  {
242  const double & min = pos_1.longitude () < pos_2.longitude () ? pos_1.longitude () : pos_2.longitude ();
243  const double & max = pos_1.longitude () > pos_2.longitude () ? pos_1.longitude () : pos_2.longitude ();
244  next_position_.longitude (madara::utility::rand_double(min, max));
245  next_position_.latitude (pos_1.latitude ());
246  }
247  else // other arbitrary line
248  {
249  const double slope = delta_lon / delta_lat;
250  next_position_.latitude (madara::utility::rand_double(pos_1.latitude (), pos_2.latitude ()));
251  next_position_.longitude (pos_1.longitude () + slope * (next_position_.latitude () - pos_1.latitude ()));
252  }
253 
254  // fill in altitude on waypoint
255  next_position_.altitude (self_->agent.desired_altitude.to_double ());
256 }
