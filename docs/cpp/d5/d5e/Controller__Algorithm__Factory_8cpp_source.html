258 {
259  algorithms::Base_Algorithm * result = 0;

263  gams::loggers::LOG_MAJOR,
264  "gams::algorithms::Controller_Algorithm_Factory::create:" \
265  " creating \"%s\" algorithm.\n", type.c_str ());
266 
267 
268  if (type != "")
269  {
270  Factory_Map::iterator it = factory_map_.find (type);
271  if (it != factory_map_.end ())
272  {
273  result = it->second->create (args, knowledge_, platform_,
274  sensors_, self_, devices_);
275  }
276  else

281  "gams::algorithms::Controller_Algorithm_Factory::create:" \
282  " could not find \"%s\" algorithm.\n", type.c_str ());
283  }
