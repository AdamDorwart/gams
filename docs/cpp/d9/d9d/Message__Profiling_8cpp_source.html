173 gams::algorithms::Message_Profiling::Message_Filter::filter (
174  Madara::Knowledge_Map& records,
175  const Madara::Transport::Transport_Context& transport_context,
176  Madara::Knowledge_Engine::Variables& /*var*/)
177 {
178  // get/construct data struct
179  Message_Data& data = msg_map_[transport_context.get_originator ()];
180 
181  // loop through each update
182  for (Madara::Knowledge_Map::const_reference update : records)
183  {
184  // we only care about specific messages
185  if (update.second.is_string_type () && update.first.find (key_prefix_) == 0)
186  {
187  // get msg number
188  size_t msg_num;
189  stringstream (update.second.to_string ()) >> msg_num;
190 
191  // is this the first?
192  if (data.first == size_t (-1))
193  {
194  data.first = msg_num;
195  data.last = msg_num;
196  }
197  // is this a previously missing message?
198  else if (msg_num < data.last)
199  {
200  // is this earlier than first?
201  if (msg_num < data.first)
202  data.first = msg_num;
203  }
204  else // this could only be a new message past last
205  {
206  data.last = msg_num;
207  }
208 
209  if (data.present.size() <= msg_num)
210  data.present.reserve (msg_num * 2);
211  data.present [msg_num] = true;
212  }
213  }
214 }
