213 gams::algorithms::area_coverage::Uniform_Random_Edge_Coverage::generate_new_position ()
214 {
215  // select new edge
216  int num_edges = int (region_.vertices.size());
217  int target_edge = Madara::Utility::rand_int (0, num_edges-1);
218 
219  // get endpoints
220  const utility::GPS_Position & pos_1 = region_.vertices[target_edge];
221  const utility::GPS_Position & pos_2 =
222  region_.vertices[(target_edge + 1) % num_edges];
223 
224  // get random point on line
225  double delta_lat = pos_2.latitude () - pos_1.latitude ();
226  double delta_lon = pos_2.longitude () - pos_1.longitude ();
227  if (delta_lon == 0) // north/south line
228  {
229  const double & min = pos_1.latitude () < pos_2.latitude () ? pos_1.latitude () : pos_2.latitude ();
230  const double & max = pos_1.latitude () > pos_2.latitude () ? pos_1.latitude () : pos_2.latitude ();
231  next_position_.latitude (Madara::Utility::rand_double(min, max));
232  next_position_.longitude (pos_1.longitude ());
233  }
234  else if (delta_lat == 0) // east/west line
235  {
236  const double & min = pos_1.longitude () < pos_2.longitude () ? pos_1.longitude () : pos_2.longitude ();
237  const double & max = pos_1.longitude () > pos_2.longitude () ? pos_1.longitude () : pos_2.longitude ();
238  next_position_.longitude (Madara::Utility::rand_double(min, max));
239  next_position_.latitude (pos_1.latitude ());
240  }
241  else // other arbitrary line
242  {
243  const double slope = delta_lon / delta_lat;
244  next_position_.latitude (Madara::Utility::rand_double(pos_1.latitude (), pos_2.latitude ()));
245  next_position_.longitude (pos_1.longitude () + slope * (next_position_.latitude () - pos_1.latitude ()));
246  }
247 
248  // fill in altitude on waypoint
