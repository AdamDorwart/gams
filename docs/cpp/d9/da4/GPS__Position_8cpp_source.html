154 void
155 gams::utility::GPS_Position::direction_to (const GPS_Position& rhs,
156  double& phi) const
157 {
163  double phi_1 = DEG_TO_RAD (latitude ());
164  double phi_2 = DEG_TO_RAD (rhs.latitude ());
165  double del_lambda = DEG_TO_RAD (rhs.longitude () - longitude ());
166  if (fabs (del_lambda) > M_PI)
167  del_lambda = (del_lambda > 0) ? (-2 * M_PI + del_lambda) : (2 * M_PI + del_lambda);
168  double del_psi = log (tan (phi_2 / 2 + M_PI / 4) / tan(phi_1 / 2 + M_PI / 4));
169  double theta = atan2 (del_lambda, del_psi);
170  theta = fmod (theta + 2 * M_PI, 2 * M_PI);
171 
172  phi = theta;
173 }

176 gams::utility::GPS_Position::distance_to (const GPS_Position & rhs) const
177 {
182  const double EARTH_RADIUS = 6371000.0;
183  const double EARTH_CIRCUMFERENCE = 2 * EARTH_RADIUS * M_PI;
184 
185  // calculate north/south distance
186  const double ns_dif = EARTH_CIRCUMFERENCE * (this->x - rhs.x) / 360.0;
187 
188  // calculate east/west distance
189  const double r_prime = EARTH_RADIUS * cos (DEG_TO_RAD (this->x));
190  const double circumference = 2 * r_prime * M_PI;
191  const double ew_dif = circumference * (this->y - rhs.y) / 360.0;
192 
193  // calculate altitude difference
194  const double alt_dif = this->z - rhs.z;
195 
196  // use distance formula
197  const double dist =
198  sqrt (pow (ns_dif, 2.0) + pow (ew_dif, 2.0) + pow (alt_dif, 2.0));
199 
200  madara_logger_ptr_log (gams::loggers::global_logger.get (),
201  gams::loggers::LOG_MAJOR,
202  "gams::utility::GPS_Position::distance_to:" \

204  rhs.to_string ().c_str (), dist);

225 gams::utility::GPS_Position::to_gps_position (
226  const Position & source, const GPS_Position & ref)
227 {
228  GPS_Position ret;
229 
230  // assume the Earth is a perfect sphere
231  const double EARTH_RADIUS = 6371000.0;
232  const double EARTH_CIRCUMFERENCE = 2 * EARTH_RADIUS * M_PI;
233 
234  // convert the latitude/x coordinates
235  ret.x = source.x * 360.0 / EARTH_CIRCUMFERENCE + ref.x;
236 
237  // assume the meters/degree longitude is constant throughout environment
238  // convert the longitude/y coordinates
239  double r_prime = EARTH_RADIUS * cos (DEG_TO_RAD (ref.x));
240  double circumference = 2 * r_prime * M_PI;
241  ret.y = source.y / circumference * 360 + ref.y;
242 
243  // keep same altitude
244  ret.z = ref.z + source.z;
245 
246  return ret;
