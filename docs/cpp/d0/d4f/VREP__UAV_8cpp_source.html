344 void
345 gams::platforms::VREP_UAV::Target_Mover::run ()
346 {
347  double local_move_speed = move_speed_.to_double () / RATE;
348 
349  // get destination
350  utility::Position dest_pos;
351  dest_pos.from_container (destination_);
352  simxFloat dest_arr[3];
353  position_to_array (dest_pos, dest_arr);
354 
355  // set current position of node target
356  simxFloat curr_arr[3];
357  position_to_array (target_pos_, curr_arr);
358 
359  // get distance to destination
360  double distance_to_destination = dest_pos.distance_to_2d (target_pos_);
361 
362  madara_logger_ptr_log (gams::loggers::global_logger.get (),
363  gams::loggers::LOG_DETAILED,
364  "gams::platforms::VREP_UAV::Target_Mover::run:" \
365  " moving to (%f,%f, distance %f m)\n",
366  dest_pos.x, dest_pos.y, distance_to_destination);
367 
368  // check if quadrotor has reached target (within epsilon)
369  if(distance_to_destination <= local_move_speed)
370  {
371  target_pos_ = dest_pos;
372  simxSetObjectPosition (client_id_, node_target_, -1, dest_arr,
373  simx_opmode_oneshot_wait);
374  return;
375  }
376 
377  // move quadrotor target closer to the desired position
378  if(distance_to_destination < local_move_speed) // we can get to target in one step
379  {
380  madara_logger_ptr_log (gams::loggers::global_logger.get (),
381  gams::loggers::LOG_DETAILED,
382  "gams::platforms::VREP_UAV::Target_Mover::run:" \
383  " moving to target instantly\n");
384 
385  curr_arr[0] = dest_arr[0];
386  curr_arr[1] = dest_arr[1];
387  curr_arr[2] = dest_arr[2];
388  }
389  else // we cannot reach target in this step
390  {
391  madara_logger_ptr_log (gams::loggers::global_logger.get (),
392  gams::loggers::LOG_DETAILED,
393  "gams::platforms::VREP_UAV::Target_Mover::run:" \
394  " calculating new target location\n");
395 
396  // how far do we have to go in each dimension
397  double dist[3];
398  for (int i = 0; i < 3; ++i)
399  dist[i] = fabs (curr_arr[i] - dest_arr[i]);
400 
401  // update target position
402  for (int i = 0; i < 3; ++i)
403  {
404  if(curr_arr[i] < dest_arr[i])
405  curr_arr[i] += dist[i] * local_move_speed / distance_to_destination;
406  else
407  curr_arr[i] -= dist[i] * local_move_speed / distance_to_destination;
408  }
409  }
410 
411  // send movement command
412  simxSetObjectPosition (client_id_, node_target_, -1, curr_arr,
413  simx_opmode_oneshot_wait);
414  array_to_position (curr_arr, target_pos_);
415 
416  madara_logger_ptr_log (gams::loggers::global_logger.get (),
417  gams::loggers::LOG_ERROR,
418  "gams::platforms::VREP_UAV::move:" \
419  " setting target to \"%f,%f,%f\"\n", curr_arr[0], curr_arr[1], curr_arr[2]);
420 }
