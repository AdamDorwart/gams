306  void from_rotation_vector(double rx, double ry, double rz)
307  {
308  double magnitude = sqrt(rx * rx + ry * ry + rz * rz);
309  if(magnitude == 0)
310  {
311  _w = 1;
312  _x = _y = _z = 0;
313  }
314  else
315  {
316  double half_mag = magnitude / 2;
317  double cos_half_mag = cos(half_mag);
318  double sin_half_mag = sin(half_mag);
319  _w = cos_half_mag;
320  _x = (rx / magnitude) * sin_half_mag;
321  _y = (ry / magnitude) * sin_half_mag;
322  _z = (rz / magnitude) * sin_half_mag;
323  }

353  static void hamilton_product(Quaternion &into, const Quaternion &lhs, const Quaternion &rhs)
354  {
355  double A = (lhs._w + lhs._x) * (rhs._w + rhs._x),
356  B = (lhs._z - lhs._y) * (rhs._y - rhs._z),
357  C = (lhs._w - lhs._x) * (rhs._y + rhs._z),
358  D = (lhs._y + lhs._z) * (rhs._w - rhs._x),
359  E = (lhs._x + lhs._z) * (rhs._x + rhs._y),
360  F = (lhs._x - lhs._z) * (rhs._x - rhs._y),
361  G = (lhs._w + lhs._y) * (rhs._w - rhs._z),
362  H = (lhs._w - lhs._y) * (rhs._w + rhs._z);
363 
364  into._w = B + (-E - F + G + H) / 2;
365  into._x = A - ( E + F + G + H) / 2;
366  into._y = C + ( E - F + G - H) / 2;
367  into._z = D + ( E - F - G + H) / 2;
368  }
