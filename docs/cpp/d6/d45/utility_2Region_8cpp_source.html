110 gams::utility::Region::contains (const GPS_Position & p) const
111 {
112  // check if in bounding box
113  if (p.latitude () < min_lat_ || p.latitude () > max_lat_ ||
114  p.longitude () < min_lon_ || p.longitude () > max_lon_)
115  {
116  return false;
117  }
118 
119  // check if point in polygon code from
120  // http://www.ecse.rpi.edu/Homepages/wrf/Research/Short_Notes/pnpoly.html
121  size_t i, j;
122  bool ret = false;
123  for (i = 0, j = vertices.size() - 1; i < vertices.size(); j = i++)
124  {
125  if ( (vertices[i].longitude () > p.longitude ()) !=
126  (vertices[j].longitude () > p.longitude ()))
127  {
128  if (p.latitude () < (vertices[j].latitude () - vertices[i].latitude ()) * (p.longitude () - vertices[i].longitude ()) /
129  (vertices[j].longitude () - vertices[i].longitude ()) +
130  vertices[i].latitude ())
131  {
132  ret = !ret;
133  }
134  }
135  }
136 
137  // check if this is a vertex point
138  if (!ret)
139  {
140  for (unsigned int i = 0; i < vertices.size() && !ret; ++i)
141  ret = (vertices[i] == p);
142  }
143 
144  // TODO: add check for border point
145 
146  return ret;

150 gams::utility::Region::distance (const GPS_Position& p) const
151 {
152  // if point is in region, then the distance is 0
153  if (contains (p))
154  return 0;
155 
156  // convert to cartesian coords with equirectangular projection
157  const GPS_Position sw (min_lat_, min_lon_);
158  vector<Position> local_vertices;
159  for (size_t i = 0; i < vertices.size (); ++i)
160  local_vertices.push_back (vertices[i].to_position (sw));
161  Position local_p = p.to_position (sw);
162 
163  // else we check for distance from each edge
164  double min_dist = DBL_MAX;
165  for (size_t i = 0; i < local_vertices.size (); ++i)
166  {
167  const size_t i_1 = (i + 1) % local_vertices.size();
168  double dist = local_vertices[i].distance_to_2d (local_vertices[i_1], local_p);
169  if (dist < min_dist)
170  min_dist = dist;
171  }
172 
173  return min_dist;

222 double
223 gams::utility::Region::get_area () const
224 {
225  if (vertices.size() < 3)
226  return 0; // degenerate polygon
227 
228  // see http://geomalgorithms.com/a01-_area.html
229  // Convert all units to cartesian
230  vector<Position> cart_vertices;
231  for (unsigned int i = 0; i < vertices.size(); ++i)
232  {
233  Position p = vertices[i].to_position (vertices[0]);
234  cart_vertices.push_back (p);
235  }
236 
237  // perform calculations with cartesian vertices
238  double area = 0.0;
239  size_t i, j, k;
240  size_t num_vertices = cart_vertices.size ();
241  for (i = 1, j = 2, k = 0; i < num_vertices; ++i, ++j, ++k)
242  {
243  area += cart_vertices[i].x *
244  (cart_vertices[j % num_vertices].y - cart_vertices[k].y);
245  }
246  area += cart_vertices[0].x * (cart_vertices[1].y - cart_vertices[num_vertices - 1].y);
247  return fabs(area / 2);
248 }
