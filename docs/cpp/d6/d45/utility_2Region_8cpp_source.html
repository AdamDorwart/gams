139 gams::utility::Region::contains (const GPSPosition & p) const
140 {
141  // check if in bounding box
142  if (p.latitude () < min_lat_ || p.latitude () > max_lat_ ||
143  p.longitude () < min_lon_ || p.longitude () > max_lon_)
144  {
145  return false;
146  }
147 
148  // check if point in polygon code from
149  // http://www.ecse.rpi.edu/Homepages/wrf/Research/ShortNotes/pnpoly.html
150  size_t i, j;
151  bool ret = false;
152  for (i = 0, j = vertices.size() - 1; i < vertices.size(); j = i++)
153  {
154  if ( (vertices[i].longitude () > p.longitude ()) !=
155  (vertices[j].longitude () > p.longitude ()))
156  {
157  if (p.latitude () < (vertices[j].latitude () - vertices[i].latitude ()) * (p.longitude () - vertices[i].longitude ()) /
158  (vertices[j].longitude () - vertices[i].longitude ()) +
159  vertices[i].latitude ())
160  {
161  ret = !ret;
162  }
163  }
164  }
165 
166  // check if this is a vertex point
167  if (!ret)
168  {
169  for (unsigned int i = 0; i < vertices.size() && !ret; ++i)
170  ret = (vertices[i] == p);
171  }
172 
173  // TODO: add check for border point
174 
175  return ret;

178 double
179 gams::utility::Region::distance (const GPSPosition& p) const
180 {
181  // if point is in region, then the distance is 0
182  if (contains (p))
183  return 0;
184 
185  // convert to cartesian coords with equirectangular projection
186  const GPSPosition sw (min_lat_, min_lon_);
187  vector<Position> local_vertices;
188  for (size_t i = 0; i < vertices.size (); ++i)
189  local_vertices.push_back (vertices[i].to_position (sw));
190  Position local_p = p.to_position (sw);
191 
192  // else we check for distance from each edge
193  double min_dist = DBL_MAX;
194  for (size_t i = 0; i < local_vertices.size (); ++i)
195  {
196  const size_t i_1 = (i + 1) % local_vertices.size();
197  double dist = local_vertices[i].distance_to_2d (local_vertices[i_1], local_p);
198  if (dist < min_dist)
199  min_dist = dist;
200  }
201 
202  return min_dist;

250 double
251 gams::utility::Region::get_area () const
252 {
253  if (vertices.size() < 3)
254  return 0; // degenerate polygon
255 
256  // see http://geomalgorithms.com/a01-_area.html
257  // Convert all units to cartesian
258  vector<Position> cart_vertices;
259  for (unsigned int i = 0; i < vertices.size(); ++i)
260  {
261  Position p = vertices[i].to_position (vertices[0]);
262  cart_vertices.push_back (p);
263  }
264 
265  // perform calculations with cartesian vertices
266  double area = 0.0;
267  size_t i, j, k;
268  size_t num_vertices = cart_vertices.size ();
269  for (i = 1, j = 2, k = 0; i < num_vertices; ++i, ++j, ++k)
270  {
271  area += cart_vertices[i].x *
272  (cart_vertices[j % num_vertices].y - cart_vertices[k].y);
273  }
274  area += cart_vertices[0].x * (cart_vertices[1].y - cart_vertices[num_vertices - 1].y);
275  return fabs(area / 2);
276 }
